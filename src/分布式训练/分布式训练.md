#### 集合通信（collective communication）
1. 广播（broadcast）

即将某个 GPU 上的数据复制发送给组内的其他节点

![](https://img.leftover.cn/img-md/202606262051626.png)

```python
import torch
import torch.distributed as dist
import os


def main():
    rank = int(os.environ["RANK"])
    local_rank = int(os.environ["LOCAL_RANK"])
    world_size = int(os.environ["WORLD_SIZE"])
    logical_cpus = os.cpu_count()

    print(f"logical_cpus:{logical_cpus}")
    os.environ["OMP_NUM_THREADS"] = str(logical_cpus // 4)
    print(f"启动进程: Rank {rank} (Local Rank {local_rank}), World Size {world_size}")

    torch.cuda.set_device(local_rank)
    device = torch.device(f"cuda:{local_rank}")
    dist.init_process_group(backend="nccl")

    input_tensor = torch.arange(rank, 8 + rank, dtype=torch.float32, device=device)
    print(f"[Rank {rank}] 执行boradcast前的数据: {input_tensor}")

    dist.barrier(device_ids=[local_rank])
    dist.broadcast(input_tensor, src=0)
    print(f"[Rank {rank}] 执行后boradcast的数据: {input_tensor}")
    dist.destroy_process_group()


if __name__ == "__main__":
    main()

```

2. scatter

![](https://img.leftover.cn/img-md/20260626211653331.png)

将一个节点上的数据切分之后发送给不同的节点

在 pytorch 中具体如何使用：

下面的代码中我们让 rank0 发送数据，其他设备的 scatter_list 传入 None 即可

```python
import torch
import torch.distributed as dist
import os

def main():
    # 当使用 torchrun 启动时，这些环境变量会被自动注入
    rank = int(os.environ["RANK"])             # 全局第几个进程
    local_rank = int(os.environ["LOCAL_RANK"]) # 当前机器上的第几个 GPU
    world_size = int(os.environ["WORLD_SIZE"]) # 总共有多少个进程

    # rank  = dist.get_rank() # 不传入group 则取得的是全局的rank，传入group，则是特定子组里面的rank
    # # local_rank = dist.get_global_rank() # 需传入组号 和 rank ，即在这个子组排号为rank的设备在全局中的rank为多少
    # world_size = dist.get_world_size()

    print(f"启动进程: Rank {rank} (Local Rank {local_rank}), World Size {world_size}")

    # [关键点 1] 必须设置当前进程使用的 GPU 设备
    # 如果不设置，所有进程可能会抢占同一块 GPU (Device 0) 导致显存OOM或冲突
    torch.cuda.set_device(local_rank)
    device = torch.device(f"cuda:{local_rank}")
    dist.init_process_group(backend="nccl")
    scatter_list = None
    if rank ==0: 
        scatter_list = [(rank+1) * torch.arange(1,3,dtype=torch.float32,device =device) *i for i in range(world_size)]
        print(f"[Rank {rank}] scatter_list: {scatter_list}")
    recv_tensor = torch.zeros(2,dtype=torch.float32,device=device)
    print(f"[Rank {rank}] 执行scatter前的recv_tensor数据: {recv_tensor}")
    dist.scatter(recv_tensor,scatter_list=scatter_list,src=0)
    print(f"[Rank {rank}] scatter 后的结果: {recv_tensor}")
    dist.destroy_process_group()

if __name__ == "__main__":
    main()
```

3. gather

所有节点将数据发送给一个特定的节点，然后将数据拼接起来

![](https://img.leftover.cn/img-md/202606262052592.png)

```python
import torch
import torch.distributed as dist
import os


def main():
    rank = int(os.environ["RANK"])
    local_rank = int(os.environ["LOCAL_RANK"])
    world_size = int(os.environ["WORLD_SIZE"])
    logical_cpus = os.cpu_count()

    print(f"logical_cpus:{logical_cpus}")
    os.environ["OMP_NUM_THREADS"] = str(logical_cpus // 4)
    print(f"启动进程: Rank {rank} (Local Rank {local_rank}), World Size {world_size}")

    torch.cuda.set_device(local_rank)
    device = torch.device(f"cuda:{local_rank}")
    dist.init_process_group(backend="nccl")

    input_tensor = torch.arange(rank, 8 + rank, dtype=torch.float32, device=device)
    print(f"[Rank {rank}] 执行gather前的数据: {input_tensor}")

    dist.barrier(device_ids=[local_rank])
    dis_rank = 0
    gather_list = None
    if rank == dis_rank:
        gather_list = [
            torch.zeros(8, dtype=torch.float32, device=device)
            for _ in range(world_size)
        ]
    dist.gather(input_tensor, gather_list=gather_list, dst=dis_rank)
    print(f"[Rank {rank}] gather_list的数据: {gather_list}")
    dist.destroy_process_group()


if __name__ == "__main__":
    main()

```

4. all-gather

所有节点都持有数据，最后所有节点都获得了所有人的数据拼接之后的结果（相当于 gather+broadcast）

![](https://img.leftover.cn/img-md/202606262052612.png)

```python

import os


def main():
    rank = int(os.environ["RANK"])
    local_rank = int(os.environ["LOCAL_RANK"])
    world_size = int(os.environ["WORLD_SIZE"])
    logical_cpus = os.cpu_count()

    print(f"logical_cpus:{logical_cpus}")
    os.environ["OMP_NUM_THREADS"] = str(logical_cpus // 4)
    print(f"启动进程: Rank {rank} (Local Rank {local_rank}), World Size {world_size}")

    torch.cuda.set_device(local_rank)
    device = torch.device(f"cuda:{local_rank}")
    dist.init_process_group(backend="nccl")

    input_tensor = torch.arange(rank, 8 + rank, dtype=torch.float32, device=device)
    print(f"[Rank {rank}] 执行all-gather前的数据: {input_tensor}")

    dist.destroy_process_group()


if __name__ == "__main__":
    main()
```

所有节点将数据发送给一个节点，并在该节点进行聚合操作**（SUM、MAX、MIN、Product（乘积））等，SUM 用的最多**

![](https://img.leftover.cn/img-md/202606262052423.png)

```python

import torch.distributed as dist
import os


def main():
    rank = int(os.environ["RANK"])
    local_rank = int(os.environ["LOCAL_RANK"])
    world_size = int(os.environ["WORLD_SIZE"])
    logical_cpus = os.cpu_count()

    print(f"logical_cpus:{logical_cpus}")
    os.environ["OMP_NUM_THREADS"] = str(logical_cpus // 4)
```

```python
import torch
import torch.distributed as dist
import os


def main():
    rank = int(os.environ["RANK"])
    local_rank = int(os.environ["LOCAL_RANK"])
    world_size = int(os.environ["WORLD_SIZE"])
    logical_cpus = os.cpu_count()

    print(f"logical_cpus:{logical_cpus}")
    os.environ["OMP_NUM_THREADS"] = str(logical_cpus // 4)
    print(f"启动进程: Rank {rank} (Local Rank {local_rank}), World Size {world_size}")

    torch.cuda.set_device(local_rank)
    device = torch.device(f"cuda:{local_rank}")
    dist.init_process_group(backend="nccl")

    input_list = [
        torch.arange(r, 8 + r, dtype=torch.float32, device=device)
        for r in range(world_size)
    ]

        print("*"*20)

    dist.reduce_scatter_tensor(reduce_scatter_tensor_output_tensor, torch.stack(input_list), op=dist.ReduceOp.SUM)

        print("*"*20)
    print(f"[Rank {rank}]执行reduce_scatter_tensor之后的reduce_scatter_tensor_output_tensor: {reduce_scatter_tensor_output_tensor}")
```

## fp16 混合精度训练时的大概流程
![](https://img.leftover.cn/img-md/202606262052159.png)



在我们使用 fp16 混合精度训练的时候，

前向传播时通常会激活值转为 fp16

最后再在反向传播的时候计算梯度（fp16）

但是为了保证准确度，在参数更新的时候，优化器相关的状态通常会采用 float32 进行存储，在参数更新的时候，将梯度转为 float32，再进行更新参数，再将更新后的参数转为 float16，进行下一轮的前向传播

## 梯度检查点
### 解释
这里查看 gemini 的[回答](https://gemini.google.com/share/4049d8839216) 。

假如我们有一个 N 层的神经网络，其前向传播大致像这样：

$ x_0 -> Layer 1 -> x_1 -> Layer2 -> ... -> Layer N -> x_N $

对于$ Layer \quad i, 其输入为 x_{i-1},输出为x_i
 $



反向传播大致像这样：

$ grad_N -> Layer N -> grad_{N-1} -> ... -> Layer1 -> grad_0 $

因此要计算 Layer i 的梯度，我们通常需要 $ grad_i $ （Layer i 层的输出的梯度）和 $ x_{i-1}
 $ （Layer i 的输入 ）

在反向传播的时候，前向传播已经结束了，因此为了在反向传播的时候拿到前向传播的时候的中间的激活值，我们通常会将所有的激活值保存在内存/显存中，以便反向传播的时候计算梯度时使用。

> 因此如果我们有 N 层，则必须存储 N 个激活值，故内存成本和模型的层数成线性关系，O(N)
>

**梯度检查点的原理：**

   plain    上面说，为了反向传播的时候能拿到中间的激活值，我们需要在前向传播的时候保存所有的激活值。而梯度检查点则是将 N 层的网络切为 m 个块（也叫 m 个检查点），每个块有 k 层；在前向传播的时候，**我们只保存每个块的第一个激活值,**因此我们有 m 个检查点，内存的空间复杂度为 O(N/k)。

当反向传播进入一个块时，我们会通过这个块的第一个激活值进行前向传播，计算出这个块的其余激活值，并临时保存下来，当这个块的反向传播计算完时，临时保存的激活值则会被丢弃。

重复这个过程，直到完成反向传播。因此综上，使用梯度检查点的时候的梯度的内存的空间复杂度为 O(N/k + k)

因此我们要选择一个 k，使得总内存 O(N/k+k)最小，

![](https://img.leftover.cn/img-md/202606262052651.png)



梯度检查点可以节省大量的显存，但代价则是增加了一次完整的前向传播



### pytorch 中使用梯度检查点
只需要在需要使用的 block 或者 layer 中，forward 的时候使用`torch.utils.checkpoint.checkpoint`来包裹对应的代码即可，如下所示

```python
from torch.utils.checkpoint import checkpoint
class CheckPointLeNet5(nn.Module):
    def __init__(self, model,use_checkpoint):
        super(CheckPointLeNet5, self).__init__()
        self.model = model
        self.use_checkpoint = use_checkpoint
    def forward(self,x):
        if self.use_checkpoint:
            return checkpoint(self.model, x,use_reentrant=False)
        else:
            return self.model(x)  

```

## 数据并行（DP）
数据并行适合一张显卡能够装下整个模型的情况，顾名思义，数据并行就是在数据维度进行划分的方式，数据并行通常有两种实现方案，一种是<font style="color:rgb(25, 27, 31);">Parameter Server 的方式，(对应于 pytorch 的 nn.DataParallel  )一种是 Ring All Reduce (对应于 pytorch 的 nn.DistributedDataParallel )</font>

### 参数服务器（<font style="color:rgb(25, 27, 31);">Parameter Server</font>）
通常我们会有两种 GPU，一种是计算 GPU（worker），一种是梯度聚合 GPU（server），worker 通常有多个，server 可以有 1 个或者多个（通常是 1 个）

![](https://img.leftover.cn/img-md/202606262052357.png)

1. 我们的模型复制到每个 worker 上，
2. 同时在训练的时候，将一个 batch 的数据平均分配到多个 gpu 上进行计算，同时反向传播计算梯度。**此时，每个 gpu 上的梯度是对应的 batch 子集的梯度，而不是整个 batch 计算出来的梯度**
3. **最后将所有的梯度汇集到一个 gpu（server）上相加起来，再广播分发到其他的 gpu 上，最后每个 gpu 上的模型同步地更新参数 （即 all-reduce 操作）**

在这种框架下，**我们的 server 通讯量很大，因为他要和所有的其他 worker 进行通信，因此 server 的带宽会称为整个系统的瓶颈**

> 为了减少通讯量，也可以使用某个 worker 作为 server，例如选择 GPU 0 作为 server
>

### Ring All-reduce
参考资料：[https://zhuanlan.zhihu.com/p/504957661](手把手推导Ring%20All-reduce的数学性质)

上面的数据并行方式采用 Parameter Server 模式 ,性能的瓶颈会在单个 GPU 上面，而 Ring All-reduce 则是将 GPU 组成一个环，我们将每个 GPU 上的梯度分为 n 块，

![](https://img.leftover.cn/img-md/202606262052934.png)

如图，第一步中，每个 GPU 都负责一块数据，并向左边的设备发送，同时接收右边发送过来的数据并累加到对应位置的数据上去；这里 GPU0 向 GPU3 发送第一块数据，同时接收 GPU1 发送过来的第二块数据并累加到对应位置

第二步 GPU 0 将第二块数据发送给左边的 GPU3，并接收 GPU1 发送过来的第三块数据并累加到对应位置



3 步过后，**每个 GPU 都有某个块的所有数据 ，这时候 GPU 需要将自己的数据发送给别的 GPU 并接收别的 GPU 的数据，同样可以使用环状通信算法实现**

假设每个 GPU 的梯度的数据量为 V，分为 p 块（有 p 个设备），则每一次单个 GPU 发送 V/p 的数据量，整个过程每个设备向外发送了$ \frac{(p-1)V}{p} $ 大小的数据，同时也接收了$ \frac{(p-1)V}{p} $大小的数据，假设带宽为$ \beta
 $,则整个过程需要的时间为$ \frac{(p-1)V}{p \beta}
 $，当 p 足够大时，时间近似为$ \frac{V}{\beta} $ ,并且这个时间和设备数 p 无关



![](https://img.leftover.cn/img-md/202606262052042.png)

和上面的过程类似，最后每个 GPU 上都有完整的梯度数据

## 模型并行
随着模型变得越来越大，一张卡已经装不下模型了，因此我们可以将模型按层切分到不同的 GPU 上，这就是**模型并行**，模型并行通常包括：**张量并行和管线（流水线）并行**

![](https://img.leftover.cn/img-md/1763449007622-f512ff8a-00e1-4502-8d7c-7989ea331557.png)

上图是最原始的模型并行的方式：

1. 我们将模型的不同层分到不同的 GPU 上（上图分到 4 个 GPU 上）
2. 然后拥有前面层的 GPU 先进行前向传播，然后把激活值传给下一个 GPU，以此类推，直到最后一个 GPU 做完 forward，再从最后一个 GPU 开始进行反向传播，并更新参数
3. 可以看出上面的并行方式特别低效，每次都只有一个 GPU 在工作

### 管线（流水线）并行
#### Gpipe
Gpipe 的思路是在模型并行的基础上，进一步引入了数据并行

![](https://img.leftover.cn/img-md/202606262052089.png)

如上图的 c 所示，我们可以将一个 batch 的数据分为 M 个（图中是 4），送入 GPU，首先 GPU 0 计算 MB0，激活值传给 GPU1，接着 GPU1 计算 MB0 （同时 GPU0 会计算 MB1），以此类推，直到最后一个 GPU 计算完$ MB_{M-1} $，

此时从最后一个 GPU 开始进行反向传播；

反向传播也是类似的思路，从最后一个 GPU 开始，先对$ MB_{M-1} $计算梯度，再将梯度的信息传给前一个 GPU，前一个 GPU 计算$ MB_{M-1}
 $的梯度 （同时最后一个 GPU 开始计算$ MB_{M-2} $的梯度，以此类推）

最后在第一个 GPU 计算完 MB0 的梯度之后，每个 GPU 将所有的 mirco-batch 的梯度相加起来，得到整个 mini-batch 的梯度，最后所有的层统一执行参数更新的操作



> GPipe 的优点在于其简单，容易理解，并且没有精度损失，也能够达到比较好的并行效果。但相比于 PipeDream 的异步的方式，其气泡时间相对较长
>



#### <font style="color:rgb(25, 27, 31);">PipeDream</font>
##### 具体流程
PipeDream 则是模型并行的方式，它将模型根据层拆分到不同的 GPU 上，如下图所示，将模型拆分到 4 个 GPU 上

![](https://img.leftover.cn/img-md/202606262053837.png)

学过计算机组成原理的读者应该知道，上面的图和计算机组成原理中的流水线的方式很像，

> 和 GPipe 的区别在于:
>
>    plain   GPipe 是一种同步的方式，即第二个 batch 前向传播的时候，需要等一个个 batch 的数据反向更新完成。 GPipe 只是在最原始的模型并行中加入了数据并行，从而提高了效率
>
>    plain   而 PipeDream 则是一张异步的方式，它可以达到很高的吞吐量，但是同时也会带来精度损失
>

图中格子中的数字表示的是第几个 batch 的数据

如上图所示，我们会有一个启动流水线的方式，进行前向传播来启动流水线，随后进行反向传播更新参数，这里有几个问题：

1. 对于上面的机器 2 的蓝色的 5 和绿色的 5，蓝色 5 表示使用第 5 个 batch 的数据进行前向传播，从图中可以看到，**这时候它的权重参数是更新了两次的 （即它是使用了更新 2 次之后的参数进行 forward），而绿色 5 表示对第 5 个 batch 的数据进行反向传播，这时候它的权重参数是更新了四次的 （即它是使用了更新 4 次之后的参数进行 backward，因为 backward 的时候会需要使用到权重参数计算梯度），这就导致了 forward 和 backward 的时候使用的权重参数不一致，导致性能出现下降。**

这里的解决思路也很简单，只需要保证 forward 和 backward 使用一样的权重参数就行，即 forward 用的那个权重参数，backward 也是有同样的权重参数，因此我们只需要每个设备上保存多个版本的权重参数即可，这样 backward 的时候就可以用到之前和 forward 时同样的权重参数

![](https://img.leftover.cn/img-md/1763452052448-346780a6-dbe3-4ec5-a8e8-73258ee4373d.png)

2. 对于上图中机器 2 的蓝色 5 和机器 3 的蓝色 5,**可以看出机器 2 forward 的时候，使用的是更新了两次之后的参数，而机器 3forward 的时候使用的是更新了 3 次之后的参数，这里使用的参数版本不一致，同样机器 1 forward 的时候是使用的是更新了 1 次之后的参数**

解决思路也很简单，我们只需要保证同一批数据 forward 时候是使用的相同版本的参数，即上面机器 2 和机器 3 在对 batch 5 进行 forward 的时候，使用更新了 1 次之后的参数即可 （跟机器 1 的参数版本保持一致）

![](https://img.leftover.cn/img-md/202606262054823.png)

##### 精度损失
从上图中可以看出，在 forward batch 2 的时候，我们本应该使用 batch 1 更新之后的参数，由于是异步的方式，这里 batch 1 数据的梯度还没计算完，这里使用的是原始的参数进行 forward，从而导致精度的损失。（理论上，即同步的方式，我们对$ batch_{n} $ 的数据进行 forward 时，我们需要使用$ batch_{n-1}
 $数据更新之后的参数）；反向传播也是同理 （backward 的时候使用的是和 forward 相同的参数来计算 grad）



## 张量并行（Megatron-LM）
reference from [https://zhuanlan.zhihu.com/p/622212228](https://zhuanlan.zhihu.com/p/622212228)

模型并行是按层将模型切开，然后放在不同的 GPU 上。而随着模型的参数量不断增加，权重矩阵 W 越来越大 （尤其在 FFN 层中的 W），单张卡已经放不下了，因此张量并行，顾名思义：我们可以将单个张量切开放在不同的 GPU 上，我们通常会将权重矩阵 W 切分为多个部分，放到不同的 GPU 上

#### 权重的拆分
1. 按行切分

![](https://img.leftover.cn/img-md/202606262054458.png)

顾名思义，按行切分则是将权重矩阵 W 按行切分，如上图所示

很明显上述的矩阵维度不对齐，不能做矩阵乘法，因此我们需要将 X 按列切分

![](https://img.leftover.cn/img-md/1763474197999-d7509296-71cd-4ae9-958d-3af8305fe973.png)

即 $ X * W = X * \begin{bmatrix}
W1
\\
W2
\end{bmatrix} = [X1,X2] * \begin{bmatrix}
W1
\\
W2
\end{bmatrix}  = X1*W1 + X2*W2 = Y1 + Y2 = Y $



![](https://img.leftover.cn/img-md/202606262054877.png)

如上图所示，我们将 W 按行切分放到 2 个 GPU 上，forward 的时候我们只需要在各自的 GPU 上分别计算 Y，最后再将输出相加

> f 操作: forward 的时候把输入切分到每个 GPU 上，即执行一个 scatter 操作，backward 的时候执行 all-gather 操作将对 X1 和对 X2 的梯度 concat 起来
>
> g 操作：forward 的时候执行 all-reduce 操作 把输入相加得到完整的 Y ，backward 的时候，由于$ \frac{\partial L}{\partial W_i} = \frac{\partial L}{\partial Y} * \frac{\partial Y}{\partial Y_i} * \frac{\partial Y_i}{\partial W_i} = \frac{\partial L}{\partial Y} *\frac{\partial Y_i}{\partial W_i}       $
>
> 因为$ \frac{\partial Y}{\partial Y_i} = 1  $
>
> 因此我们只需要将 L 对 Y 的梯度分发到不同的 GPU 上即可求得对 W_i 的梯度，因此这个 backward 的时候做的是一个<font style="color:rgb(25, 27, 31);">broadcast 操作</font>
>

2. 按列切分

![](https://img.leftover.cn/img-md/1763554244522-e7b3d1b3-d458-4187-86c6-14fbbe9297e2.png)

如上图所示，我们将权重 W 按列切分，这里维度是对齐的，X 可以直接计算，不需要再切分了

$ X * W = X *[W1,W2] = [XW1,XW2] = Y $

![](https://img.leftover.cn/img-md/202606262054524.png)

forward 的时候 X 会分别复制到不同的 GPU 上，与切分之后的权重相乘，得到 Y1 和 Y2，最后 g 执行 all-gather 操作，把张量 concat 起来

> 因此我们可以根据上图得到
>
> f 函数：forward 的时候将 X 分发到多个设备上，即执行一个<font style="color:rgb(25, 27, 31);">broadcast 操作，backward 的时候，则执行 all-reduce 操作，把每个设备上的关于 X 的梯度相加起来</font>
>
> <font style="color:rgb(25, 27, 31);">g 函数：forward 的时候将所有设备上的 Y concat 起来，即执行一个 all-gather 操作，</font>
>
> <font style="color:rgb(25, 27, 31);">因为</font>$ \frac{\partial L}{\partial Y} =  [\frac{\partial L}{\partial Y_1} ,\frac{\partial Y}{\partial Y_2} ]
  $<font style="color:rgb(25, 27, 31);">,因此 backward 的时候需要将关于 Y 的梯度切分成两部分，即做一个 scatter 操作</font>
>

#### <font style="color:rgb(25, 27, 31);">megatron 对 MLP 的切分</font>
![](https://img.leftover.cn/img-md/1763557062674-dac2cd81-28f0-489b-9d8d-3597769736f2.png)

MLP 层大致如上图所示

![](https://img.leftover.cn/img-md/202606262054616.png)

magatron 中的拆分如上图所示，对 A 采用列切分，对 B 采用行切分，由于 gelu 是逐元素计算的，并不会改变张量形状，这里我们暂时忽略它，因此得到了下列的公式

$ X*A*B = X* [A1,A2] * \begin{bmatrix}
B1
\\
B2
\end{bmatrix}  = [XA1,XA2]*\begin{bmatrix}
B1
\\
B2
\end{bmatrix}  = XA1B1 + XA2B2  = Z1 + Z2 = Z $

总体流程：

> forward：首先把 X 拷贝到两个 GPU 上个，再执行对应的计算得到 Z1 和 Z2，最后把 Z1 和 Z2 相加得到 Z
>
> backward：因为$ \frac{\partial Z}{\partial Z_1} = 1   $$ \frac{\partial Z}{\partial Z_2} = 1   $ ，因此$ \frac{\partial L}{\partial Z} =  \frac{\partial L}{\partial Z_1} = \frac{\partial L}{\partial Z_2}
      $，所以 backward 的时候首先把$ \frac{\partial L}{\partial Z}   $拷贝到两个 GPU 上，再分别在各自的 GPＵ上求梯度即可，最后求得$ \frac{\partial L}{\partial X}1  和 \frac{\partial L}{\partial X}2  $，再相加得到$ \frac{\partial L}{\partial X} $
>

因此

> f 函数：forward 的是进行<font style="color:rgb(25, 27, 31);">broadcast 操作，backward 时进行 all-reduce 操作</font>
>
> <font style="color:rgb(25, 27, 31);">g 函数：forward 时进行 all-reduce 操作，backward 时进行 broadcast 操作</font>
>

### zero-dp
在训练的时候，GPU 通常会需要存储以下的内容

![](https://img.leftover.cn/img-md/202606262054030.png)

而模型状态相关的是必须存储的

1. 优化器状态：以 Adam 为例，通常需要存储 momentum 和 variance
2. 梯度：每个参数的梯度
3. <font style="color:black;">parameters：模型参数</font>

![](https://img.leftover.cn/img-md/1763384548558-b8f3a159-1081-46b6-90a2-fc56b9ba8e6f.png)

上面我们讲到混合精度训练，首先我们会存储一份 parameter、momentum、variance

forward 之前将 float32 拷贝一份转为 float16，然后使用 float16 的参数进行 forward，得到的激活也是 float16，再 backward 计算梯度（float16），最后使用 float32 的 parameter、momentum、variance + float16 的梯度去更新模型参数。

因此我们假设有 A 个参数，则必存的参数有 parameter、momentum、variance ，大小为 4$ \Phi  $ + 4$ \Phi  $+ 4$ \Phi  $ ,中间过程需要存储的参数为 float16 的参数（2$ \Phi  $） + 梯度（2$ \Phi  $），还有激活值（激活与 batch size 有关，激活不是必须要存储的，可以使用检查点 +多做一次 forward 得到）



#### zero 1 
整个训练的时候，我们在数据并行中会分别存储一份模型参数、优化器状态，显然这部分存储会有冗余，而 zero1 则是在数据并行的基础上，对优化器的状态再进一步切分，即每个 GPU 存储一部分，需要的时候再从别的 GPU 上拿过来即可（并且优化器状态只会在更新模型参数的时候用到）

![](https://img.leftover.cn/img-md/202606262054202.png)

整体流程：

将整个 batch 的数据分成 3 分，进行 forward + backward ，然后对梯度做 all-reduce，单卡通信量 2$ \Phi  $，这样每个 GPU 上就有了整个 batch 的完整梯度，然后对 float32 的 W 进行更新，因为每个 GPU 只存储部分优化器状态，因此这里只更新部分参数

![](https://img.leftover.cn/img-md/202606262054502.png)

最后进行 all-gather 操作，把别的 GPU 上更新好的参数取过来即可（单卡通信量为$ \Phi  $）

![](https://img.leftover.cn/img-md/1763731841816-d3e82b78-d698-4e87-aa94-9bb3d4dc8d71.png)

因此我们普通的 DP 的需要的显存为（2+2+K)$ \Phi  $

> 中间过程的梯度 2 $ \Phi $，float16 的参数 2$ \Phi  $，若使用 adam 优化器，则 K 为 16
>

而 zero0 需要的显存为$ (2+2+\frac{K}{N_d})\Phi  $,N_d 为显卡数量，因为模型的状态是采用 float32 存储的，因此这种方式下我们可以降低很多显存的开销



#### zero2
同样，我们也可以将梯度拆分开来

![](https://img.leftover.cn/img-md/202606262054171.png)

具体流程：

1. 首先将数据分为 3 份，分别进行 forward 和 backward，计算梯度，这时候每个 GPU 我们得到了当前 mirco-batch 对应的梯度，即绿色+白色部分
2. 然后我们做一个 reduce-scatter 操作，将梯度 reduce 到一个 GPU 上（得到的是整个 mini-batch 的梯度）再执行 scatter 操作，最终每个 GPU 只存储部分梯度（即绿色的部分）
3. 然后使用对应的 O 和 G 去更新 float32 的 W，再对 float32 的 W 做一个 all-gather 操作，将其他 GPU 上算好的 float32 的 W 同步过来

![](https://img.leftover.cn/img-md/202606262054926.png)

跟 zero1 相比，又进一步降低了每个 GPU 上的梯度相关的显存



#### zero3
zero3 则是在 zero2 的基础上，对 float16 的参数再进一步细分，每个 GPU 上只存储部分 float16 的参数

![](https://img.leftover.cn/img-md/202606262054422.png)

因为参数在 forward 和 backward 的时候都会用到，所以 zero3 会比较复杂

具体流程：

1. 首先将数据分为 3 份，每个 GPU 一份
2. forward 的时候，做一次 all-gather，取回别的 GPU 上的 float16 的 W，得到完整的 float16 的 W 进行 forward，完成之后将不是自己的那部分 float16 的 W 丢掉
3. backward 的时候再做一次 all-gather，取回别的 GPU 上的 float16 的 W 来计算梯度，backward 做完再将不是自己的那部分 float16 的 W 丢掉
4. 之后再和 zero2 一样，做一个 reduce-scatter 操作，梯度 reduce 到一个 GPU 上（得到的是整个 mini-batch 的梯度）再执行 scatter 操作，最终每个 GPU 只存储部分梯度（即绿色的部分）
5. 每个 GPU 使用自己的 O 和 G 来更新自己的 float32 的 W，因为只维护部分的 W，因此最后不需要再对 W 进行 all-gather 操作

### ZeRO-Offload
<font style="color:rgb(0, 0, 0);">ZeRO-Offload 的思路是将要存储的大头 offload 到 CPU 上，计算的部分则留到 GPU 上</font>

+ <font style="color:rgb(0, 0, 0);">forward 和 backward 计算量高，因此和他们相关的部分，即 float16 的参数，激活，放到 GPU 上</font>
+ <font style="color:rgb(0, 0, 0);">参数更新部分计算量低，因此和他们相关的部分放入 CPU 中计算，例如 float32 的 W，优化器的状态，float16 的梯度</font>

![](https://img.leftover.cn/img-md/202606262054690.png)

大致的流程如图所示：

1. 将 fp32 的参数转为 fp16
2. 使用 fp16 的参数，在 GPU 中进行 forward 和 backward
3. 得到的 fp16 梯度 offload 到 CPU 上进行参数更新

```python
import torch
import torch.nn as nn

class LeNet5(nn.Module):
    def __init__(self):
        super(LeNet5, self).__init__()
        # 输入: 1通道, 28x28
        
        # C1 卷积层: 1 -> 6通道, 5x5核
        self.conv1 = nn.Conv2d(in_channels=1, out_channels=6, kernel_size=5)
        
        # S2 池化层: 2x2 最大池化
        self.pool = nn.MaxPool2d(kernel_size=2, stride=2)
        
        # C3 卷积层: 6 -> 16通道, 5x5核
        self.conv2 = nn.Conv2d(in_channels=6, out_channels=16, kernel_size=5)
        
        # 全连接层
        # 这里的输入维度计算：
        # 输入 28x28 -> conv1(5x5) -> 24x24 -> pool(2x2) -> 12x12
        # -> conv2(5x5) -> 8x8 -> pool(2x2) -> 4x4
        # 最终特征图大小为 16通道 * 4 * 4
        self.fc1 = nn.Linear(16 * 4 * 4, 120)
        self.fc2 = nn.Linear(120, 84)
        self.fc3 = nn.Linear(84, 10) # 10个类别

    def forward(self, x):
        # 卷积 -> 激活 -> 池化
        x = self.pool(torch.relu(self.conv1(x)))
        x = self.pool(torch.relu(self.conv2(x)))
        
        # 展平 (Flatten)
        x = x.view(-1, 16 * 4 * 4)
        
        # 全连接层
        x = torch.relu(self.fc1(x))
        x = torch.relu(self.fc2(x))
        x = self.fc3(x) # 输出层不需要激活，CrossEntropyLoss 会处理 Softmax
        return x


def spilt_batch (X,y,devices):
    X_shards,y_shards = [],[]
    for d in devices:
        X_shards.append(X.to(d))
        y_shards.append(y.to(d))
    return X_shards,y_shards    


def all_reduce_gradients(model):
    for param in model.parameters():
        if param.requires_grad and param.grad is not None:
            dist.all_reduce(param.grad.data, op=dist.ReduceOp.SUM)

def train_model(num_epochs,is_distributed):
    if is_distributed:
        rank = int(os.environ["RANK"])             # 全局第几个进程
        local_rank = int(os.environ["LOCAL_RANK"]) # 当前机器上的第几个 GPU
        world_size = int(os.environ["WORLD_SIZE"]) # 总共有多少个进程
        torch.cuda.set_device(local_rank)
        device = torch.device(f"cuda:{local_rank}")
    else:
        device = torch.device("cuda" if torch.cuda.is_available() else "cpu")

    train_set = torchvision.datasets.FashionMNIST(
        root='./data', train=True, download=True, transform=transform)
    if is_distributed:
        # 使用 DistributedSampler 确保每个进程拿到不同的数据切片
        sampler = DistributedSampler(train_set, num_replicas=world_size, rank=rank,shuffle=True)
        train_loader = DataLoader(train_set, batch_size=64, sampler=sampler, shuffle=False, num_workers=2)
    else:
        train_loader = DataLoader(train_set, batch_size=64, shuffle=True, num_workers=2)    
    classes = ('T-shirt/top', 'Trouser', 'Pullover', 'Dress', 'Coat',
            'Sandal', 'Shirt', 'Sneaker', 'Bag', 'Ankle boot')
    model = LeNet5().to(device)
    if is_distributed:
    # 为了保证所有初始化的参数一致，需要从 rank 0 广播参数到所有其他进程
        for param in model.parameters():
            dist.broadcast(param.data, src=0)
        dist.barrier()
    criterion = nn.CrossEntropyLoss()
    optimizer = optim.Adam(model.parameters(), lr=0.001)

    print(f"\nStarting training for {num_epochs} epochs...")
    start_time = time.time()
    
    for epoch in range(num_epochs):
        model.train()
        running_loss = 0.0
        
        for i, data in enumerate(train_loader):
            inputs, labels = data
            inputs, labels = inputs.to(device), labels.to(device)

            optimizer.zero_grad()
            outputs = model(inputs)
            loss = criterion(outputs, labels)
            loss.backward()
            if is_distributed:
                # all-reduce 聚合梯度
                all_reduce_gradients(model)
            optimizer.step()
            running_loss += loss.item()

            if (i+1) % 100 == 0:
                print(f'[Epoch {epoch + 1}, Batch {i + 1}] Loss: {running_loss / 100:.3f}')
                running_loss = 0.0

    print(f'Finished Training in {time.time() - start_time:.2f}s')
    if is_distributed:
        dist.barrier()
    evaluate_model(model,device,is_distributed)

    if is_distributed:
        test_sampler = DistributedSampler(test_set, num_replicas = 4,shuffle=False)

        test_loader = DataLoader(test_set, batch_size=64, shuffle=False, num_workers=2)

    correct = torch.tensor(0,device=device)
```

上面的代码主要实现了使用 resnet34 来训练 CIFAR-10,主要的关键步骤在于

1. 创建进程组（ddp_setup）和销毁进程组（ddp_cleanup）
2. 下载数据集的时候只让 rank0 下载
3. 使用 DistributedSampler 对数据进行分片，来保证每一个 GPU 采样到不同的数据
4. 在分布式训练中，每个 GPU 会计算他们对应的均值和方差，而 BN 的归一化和 batch size 密切相关，尤其是当 batch size 较小时，BN 的效果会变差，而`SyncBatchNorm`可以在计算 BN 的时候执行 all-gather 操作，得到所有 GPU 上的均值和方差，并计算全局的均值和方差，同时更新`running_mean` 和 `running_var` 来保证 BN 的行为和单卡的一致
5. 使用 ddp_model = DDP(model, device_ids=[local_rank]) 来包裹原来的 model，他会在 backward 之后自动执行 all-reduce 来聚合梯度，并且在初始化的时候会使用 broadcast，来保证每个 GPU 上模型初始化的参数一致
6. 保存模型的时候只需要使用某个 GPU 保存即可，因为每个 GPU 上的模型参数是一致的

## FSDP1 和 deepspeed 配置的解析
### FSDP1 的一些配置
1. wrap 策略：通常有`size_based_auto_wrap_policy` 、`transformer_auto_wrap_policy`

> `size_based_auto_wrap_policy`  即将大于`min_num_params` 参数的 layer 合并到一起进行分片
>
> `transformer_auto_wrap_policy` 通常则是对每个 transformer block 进行分片
>

2. 分片策略：
+ 默认的分片策略为 FULL_SHARD，即 zero3
+  SHARD_GRAD_OP 为 zero2
+  NO_SHARD 不分片，退化为 DDP
+ HYBRID_SHARD: 在单个节点上做 FULL_SHARD （适合多机多卡）
+ _HYBRID_SHARD_ZERO2:在单个节点上做 SHARD_GRAD_OP （适合多机多卡）
3. CPUOffload: 可以将参数和梯度 offload 到 CPU 上，类似于 deepspeed-offload，由于参数的更新并不是特别消耗计算资源，因此可以 offload 到 cpu 上进行，可以进一步节省显存
4. sync_module_states：为 true，则初始化的时候会将 rank0 上的参数广播给其他 rank
5. `mixed_precision` ：可以开启混合精度训练，并配置参数、梯度、缓冲区对应的数据类型
6. `backward_prefetch`： 

![](https://img.leftover.cn/img-md/202606262057257.png)

+ `BACKWARD_PRE` ：在计算第 N 层的梯度之前，就 prefetchN-1 层的参数，可以重叠通信的开销和梯度的计算，以提高速度，**缺点就是会增加峰值的显存，峰值的时候同时存在当前层的参数、梯度、下一层的参数**
+ `BACKWARD_POST` ：在计算完当前层的梯度时，prefetch 第 N-1 层的参数，相比`BACKWARD_PRE` ，其峰值显存会更低，但速度的提升也没`BACKWARD_PRE` 那么多。显存峰值的时候会同时持有当前层的梯度、下一层的参数
7. limit_all_gathers：如果为 true，则会限制同时进行 all-gather 操作，防止显存占用突然上升导致 OOM
8. use_orig_params：默认为 false，如果为 true，FSDP 则会向外暴露原始的参数结构，建议为 true

## 其他
1. 在使用了数据并行的条件下 （DP / DDP / Gpipe），通常在计算 batch normalization 的时候会有影响，因此 BN 的是在 batch 的维度进行归一化，我们数据并行的时候通常会将一个 mini-batch 划分为多个 mirco-batch，因此 batch_size 则变小了，其计算 BN 的结果也不一致，尤其是在每个 GPU 上的 batch_size 比较小的时候，会比较影响性能

> 解决方法：<font style="color:rgb(36, 36, 36);">SyncBatchNorm</font>
>
